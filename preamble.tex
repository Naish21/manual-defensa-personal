\usepackage{titlesec}

% --- Capítulos: número grande y ligero, título amplio y filete fino --------
\titleformat{\chapter}[display]
  {\normalfont\sffamily\color{primario}}
  {\filright
   {\color{secundario}\sffamily\bfseries\fontsize{15}{15}\selectfont
    \MakeUppercase{\chaptertitlename}}\;
   {\color{primario!35}\fontsize{60}{60}\selectfont\thechapter}}
  {4pt}
  {\Huge\bfseries}
  [\vspace{8pt}{\color{primario!45}\titlerule[1.1pt]}]
\titlespacing*{\chapter}{0pt}{24pt}{24pt}

% Capítulos sin numeración (índice, bibliografía): sobrio y en línea
\titleformat{name=\chapter,numberless}[display]
  {\normalfont\sffamily\color{primario}}
  {}{0pt}
  {\Huge\bfseries}
  [\vspace{8pt}{\color{primario!45}\titlerule[1.1pt]}]

\titleformat{\section}
  {\normalfont\sffamily\large\bfseries\color{primario}}
  {\thesection}{0.8em}{}
